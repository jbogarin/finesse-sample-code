 27 	getCollection: function () {
 28 		var id, reasonCode;
 29

 31 		for (id in this._collection)

 33 			if (this._collection.hasOwnProperty(id))
 34 			{

 37 				{
 38 					delete this._collection[id];
 39 				}
 40 			}
 41 		}
 42 
 43 		return this._collection;
 44 	}
